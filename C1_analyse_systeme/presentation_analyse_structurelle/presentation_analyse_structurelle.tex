\subsection{Travail demandé}

\subsubsection{Expérimenter le système pluritechnique}

\activite{Pour chacun des systèmes, suivre le document associé avec la démarche de mise en oeuvre et répondre aux différentes questions.}

\activite{Analyse des systèmes \footnote{Cette activité a pour but de vous aider à construire une synthèse.} 
\begin{enumerate}
\item Décrire le besoin et les exigences du système en s'appuyant sur un diagramme SysML approprié.
\item Définir la frontière d'étude des systèmes.
\item Identifier les grandeurs d'entrée et de sortie des systèmes.
\item Synthétiser le résultat des mesures à l'aide d'un raisonnement autour des écarts entre les performances attendues et mesurées.
\end{enumerate}
}

\subsubsection{Analyse la structure du système}
